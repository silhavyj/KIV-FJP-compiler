174  EIP = stackMemory[ESP + 4];
175  EBP = stackMemory[ESP + 3];

182  // x + y
183  case FJP::OPRType::OPR_PLUS:
184  ESP--;

186  FJP::exitProgramWithError(FJP::RuntimeErrors::ERROR_02, ERROR_CODE);
187  }
188  stackMemory[ESP] = stackMemory[ESP] + stackMemory[ESP + 1];
189  break;
190  // x - y
191  case FJP::OPRType::OPR_MINUS:
192  ESP--;
193  stackMemory[ESP] = stackMemory[ESP] - stackMemory[ESP + 1];
194  break;
195  // x * y
196  case OPR_MUL:
197  ESP--;

199  FJP::exitProgramWithError(FJP::RuntimeErrors::ERROR_02, ERROR_CODE);
200  }
201  stackMemory[ESP] = stackMemory[ESP] * stackMemory[ESP + 1];
202  break;
203  // x / y

208  }

128  std::string tokenValue;
129 
130  // Skip all comments if there are any.
131  skipComments();
132 
133  // Go through all the keywords in descending order (by their length)
134  for (auto &keyword : keywords) {
135  // Check if the current keyword would fit into the file (it doesn't exceed the file length)
136  if (currentCharIndex + keyword.first.length() > fileContent.length())
137  continue;
138 
139  // Check if it's indeed the keyword.
140  tokenValue = fileContent.substr(currentCharIndex, keyword.first.length());
141  if (tokenValue == keyword.first) {
142 
143  // Move on by the keyword length in the input file.
144  currentCharIndex += tokenValue.length();
145 
146  // Check if the keyword is not just a prefix of a custom
147  // identifier such as a variable or an array.
148  if (alphabetic_keywords.count(tokenValue)) {
149  if (currentCharIndex < static_cast<long>(fileContent.length()) && is_part_of_identifier(fileContent[currentCharIndex])) {
150  currentCharIndex -= tokenValue.length();
151  continue;
152  }
153  }
154

156  // and return the token.
157  skipWhiteCharacters();
158  return {

160  tokenValue, // value of the token
161  currentLineNumber // number of the line the token is on

163  }
164  }

169  // until you find a non-digit character.

173  if (atoi(number.c_str()) < 0) {
174  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_02, ERR_CODE, currentLineNumber);
175  }
176

178  skipWhiteCharacters();

181  number, // value of the token
182  currentLineNumber // number of the line the token is on
183

185  }

249 void FJP::Lexer::skipComments() {
250  // First off, skip all possible white characters.
251  skipWhiteCharacters();
252 
253  std::string commentStart(COMMENT_START);
254  std::string commentEnd(COMMENT_END);
255 
256  // Make sure we're not at the end of the file (that there's still enough room for a comment).
257  if (currentCharIndex + commentStart.length() >= fileContent.length())
258  return;
259 
260  // Check if the next two characters make up the start sequence of a comment.
261  std::string characterPair = fileContent.substr(currentCharIndex, commentStart.length());
262  if (characterPair != commentStart)
263  return;
264 
265  // Keep counting open comments.
266  int openComments = 1;
267 
268  currentCharIndex += commentStart.length();
269  while (openComments > 0) {
270  // Skip all white spaces.
271  skipWhiteCharacters();
272 
273  // If we reach the end of the file and there's still an unclosed comment, it's an error.
274  if (currentCharIndex + commentStart.length() > fileContent.length()) {
275  FJP::exitProgramWithError(FJP::CompilationErrors::ERROR_01, ERR_CODE);
276  }
277 
278  // Parse the next two characters.
279  characterPair = fileContent.substr(currentCharIndex, commentStart.length());
280 
281  // Check if it's an opening of another comment,
282  // closing sequence, or neither.
283  if (characterPair == commentStart) {
284  openComments++;
285  currentCharIndex += commentStart.length();
286  } else if (characterPair == commentEnd) {
287  openComments--;
288  currentCharIndex += commentStart.length();
289  } else {
290  currentCharIndex++;
291  }
292  }
293  // Skip all white spaces.
294  skipWhiteCharacters();
295 }

38 FJP::Symbol FJP::SymbolTable::findSymbol(const std::string &name) const {
39  // Default symbol that will be returned if the symbol has not been found - symbol type = SYMBOL_NOT_FOUND.
40  // We also use this symbol to find the searched symbol - the name is set to the searched one.
41  FJP::Symbol symbol = { FJP::SymbolType::SYMBOL_NOT_FOUND, name, 0, 0, 0, 0};
42 
43  // Go through all the frames (from the last one back to the first one) and for each
44  // frame, check if it holds the searched symbol. Once the symbol is found, return it.
45  for (auto iter = frames.rbegin(); iter != frames.rend(); ++iter) {
46  auto symbolIt = iter->symbols.find(symbol);
47  if (symbolIt != iter->symbols.end()) {
48  return *symbolIt;
49  }
50  }
51  // Symbol has not been found.
52  return symbol;
53 }

60 void FJP::SymbolTable::makeArray(const std::string &name, int size) {
61  FJP::Symbol symbol = { FJP::SymbolType::SYMBOL_NOT_FOUND, name, 0, 0, 0, 0};
62 
63  // Go through all the frames in a reverse order.
64  for (auto iter = frames.rbegin(); iter != frames.rend(); ++iter) {
65  // Check if the current frame holds the symbol we're searching for.
66  auto symbolIt = iter->symbols.find(symbol);
67  if (symbolIt != iter->symbols.end()) {
68  // Copy the symbol into a temporary one and set its new size
69  // which happens to be the number of elements in the array.
70  FJP::Symbol updatedSymbol = *symbolIt;
71  updatedSymbol.size = size;
72 
73  // Change the symbol type from int to int[]
74  // or from bool to bool[].
75  switch (updatedSymbol.symbolType) {
76  case FJP::SymbolType::SYMBOL_INT:
77  updatedSymbol.symbolType = FJP::SymbolType::SYMBOL_INT_ARRAY;
78  break;

82  default:
83  break;
84  }
85  // Remove the original symbol from the frame and replace it
86  // with the updated one.
87  iter->symbols.erase(*symbolIt);

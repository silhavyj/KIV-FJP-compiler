93 void FJP::Parser::processBlock() {
94  // Default allocation size of the frame - 0, base(l, EBP), EBP, EIP.
95  int frameVariableCount = FRAME_INIT_VAR_COUNT;
96 
97  // Address of the next variable that will be created within the frame.
98  nextFreeAddress = FRAME_INIT_VAR_COUNT;
99  symbolTable.createFrame();
100 
101  // Added a new instruction that will allocate a certain amount
102  // of variable on the stack. The particular value will be specified
103  // once the block has been completely parsed.
104  generatedCode.addInstruction({FJP::OP_CODE::INC, 0, 0});
105  int incInstructionAddress = generatedCode.getSize() - 1;
106 
107  // Process all const and variable declarations.
108  processConst();
109  processVariable(frameVariableCount);
110 
111  // Add a JMP instruction, so we can skip all functions and jump straight
112  // at the first instruction of the "main function".
113  int jmpAddress = generatedCode.getSize();
114  generatedCode.addInstruction({FJP::OP_CODE::JMP, 0, 0});
115 
116  // Process all functions.
117  processFunction();
118 
119  // Now we know the first address of the actual program,
120  // so we can set the jump address of the instruction that we added before.
121  generatedCode[jmpAddress].m = generatedCode.getSize();
122 
123  // Also, since all variables must have been already declared, we can
124  // specifically say how many variables need to be allocated on the stack.
125  generatedCode[incInstructionAddress].m = frameVariableCount;
126 
127  // Process a statements.
128  processStatement();
129 
130  // Destroy the frame since we're about to return from the function.
131  nextFreeAddress -= FRAME_INIT_VAR_COUNT;
132  symbolTable.destroyFrame();
133 
134  // Add a return operation as a return from the function.
135  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_RET});
136 }

464  // it is treated as a label (goto).

467  if (variable.symbolType == FJP::SymbolType::SYMBOL_NOT_FOUND) {
468  // Process the label and return the function.
469  processLabel(token.value);
470  return;
471  }
472 
473  bool isAnotherAssign = true;
474  switch (variable.symbolType) {
475  // If the identifier is an integer or bool.
476  case FJP::SymbolType::SYMBOL_INT:
477  case FJP::SymbolType::SYMBOL_BOOL:
478  // ':='
479  token = lexer->getNextToken();
480  if (token.tokenType != FJP::TokenType::ASSIGN) {
481  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_14, ERR_CODE, token.lineNumber);
482  }
483  // <identifier>
484  token = lexer->getNextToken();
485  if (token.tokenType == FJP::TokenType::IDENTIFIER) {
486

488  // not an identifier, it has to be treated as an expression.
489  // For example <identifier> := <identifier> or <identifier> := <identifier> + 55
490  token = lexer->getNextToken();
491  if (token.tokenType != FJP::TokenType::ASSIGN) {
492  lexer->returnToPreviousToken();
493  lexer->returnToPreviousToken();
494  token = lexer->getNextToken();
495  isAnotherAssign = false;
496  }

499  processAssignment();
500

502  generatedCode.addInstruction({FJP::OP_CODE::LOD, lastProcessVariable.level, lastProcessVariable.address});

504  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, 0 });
505  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_NEQ });
506  }

508  }
509  }

513

516  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, 0 });
517  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_NEQ });
518  }

520  lastProcessVariable = variable;
521  break;
522 
523  // If the symbol is an array.
524  case FJP::SymbolType::SYMBOL_INT_ARRAY:
525  case FJP::SymbolType::SYMBOL_BOOL_ARRAY:
526  // '['
527  token = lexer->getNextToken();
528  if (token.tokenType != FJP::TokenType::LEFT_SQUARED_BRACKET) {
529  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_17, ERR_CODE, token.lineNumber);
530  }
531

533  token = lexer->getNextToken();
534  processExpression();
535

537  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, variable.address });
538  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_PLUS });
539

542  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_16, ERR_CODE, token.lineNumber);
543  }
544 
545  // ':='

547  if (token.tokenType != FJP::TokenType::ASSIGN) {
548  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_14, ERR_CODE, token.lineNumber);
549  }
550

552  token = lexer->getNextToken();
553  processExpression();
554

556  if (variable.symbolType == FJP::SymbolType::SYMBOL_BOOL_ARRAY) {
557  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, 0 });
558  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_NEQ });
559  }

561  break;

564  break;
565  }

879 // foreach (<identifier> : <identifier) <statement>
880 void FJP::Parser::processForeach() {
881  // foreach
882  if (token.tokenType != FJP::TokenType::FOREACH) {
883  return;
884  }
885 
886  // '('
887  token = lexer->getNextToken();
888  if (token.tokenType != FJP::TokenType::LEFT_PARENTHESIS) {
889  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_12, ERR_CODE, token.lineNumber);
890  }
891 
892  // <identifier>
893  token = lexer->getNextToken();
894  if (token.tokenType != FJP::TokenType::IDENTIFIER) {
895  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_24, ERR_CODE, token.lineNumber);
896  }
897 
898  // Make sure the identifier exists in the symbol table.
899  FJP::Symbol iterVariable = symbolTable.findSymbol(token.value);
900  if (iterVariable.symbolType == FJP::SymbolType::SYMBOL_NOT_FOUND) {
901  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_45, ERR_CODE, token.lineNumber);
902  }
903 
904  // Make sure the identifier is one of the supported data types.
905  switch (iterVariable.symbolType) {
906  case FJP::SymbolType::SYMBOL_INT:
907  case FJP::SymbolType::SYMBOL_BOOL:
908  break;
909  default:
910  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_32, ERR_CODE, token.lineNumber);
911  }
912 
913  // ':'
914  token = lexer->getNextToken();
915  if (token.tokenType != FJP::TokenType::COLON) {
916  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_11, ERR_CODE, token.lineNumber);
917  }
918 
919  // <identifier>
920  token = lexer->getNextToken();
921  if (token.tokenType != FJP::TokenType::IDENTIFIER) {
922  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_24, ERR_CODE, token.lineNumber);
923  }
924 
925  // Make sure the identifier exists in the symbol table.
926  FJP::Symbol dataArray = symbolTable.findSymbol(token.value);
927  if (dataArray.symbolType == FJP::SymbolType::SYMBOL_NOT_FOUND) {
928  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_09, ERR_CODE, token.lineNumber);
929  }
930 
931  switch (iterVariable.symbolType) {
932  // If the type of the iterator is an integer. The array has to be an array of integers.
933  case FJP::SymbolType::SYMBOL_INT:
934  if (dataArray.symbolType != FJP::SymbolType::SYMBOL_INT_ARRAY) {
935  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_42, ERR_CODE, token.lineNumber);
936  }
937  break;
938 
939  // If the type of the iterator is bool. The array has to be an array of booleans as well.
940  case FJP::SymbolType::SYMBOL_BOOL:
941  if (dataArray.symbolType != FJP::SymbolType::SYMBOL_BOOL_ARRAY) {
942  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_42, ERR_CODE, token.lineNumber);
943  }
944  break;
945  default:
946  break;
947  }
948 
949  // ')'
950  token = lexer->getNextToken();
951  if (token.tokenType != FJP::TokenType::RIGHT_PARENTHESIS) {
952  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_13, ERR_CODE, token.lineNumber);
953  }
954  token = lexer->getNextToken();
955 
956  // Create a temporary "variable" on the stack that will be used to store
957  // the current index as we iterate through the array.
958  int indexAddress = nextFreeAddress;
959  generatedCode.addInstruction({FJP::OP_CODE::INC, 0, 1});
960 
961  // Initialize the current index to 0 (the first element of the array).
962  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, 0});
963  generatedCode.addInstruction({FJP::OP_CODE::STO, 0, indexAddress});
964 
965  // Start of the condition of the foreach loop (checking whether we've reached
966  // the end of the array or not).
967  int startForeachBody = generatedCode.getSize();
968 
969  // Check if we have just reached the end of the array (current index == array.size).
970  generatedCode.addInstruction({FJP::OP_CODE::LOD, 0, indexAddress});
971  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, dataArray.size});
972  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_NEQ});
973 
974  // Skip the body of the foreach loop if the end of the array has been reached.
975  int exitForeachAddress = generatedCode.getSize();
976  generatedCode.addInstruction({FJP::OP_CODE::JPC, 0, 0});
977 
978  // Load array[index] to the top of the stack. We need to take the base address
979  // of the array and add the current index to it, so we get the address of
980  // the current element.
981  generatedCode.addInstruction({FJP::OP_CODE::LOD, 0, indexAddress});
982  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, dataArray.address});
983  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_PLUS});
984  generatedCode.addInstruction({FJP::OP_CODE::LDA, symbolTable.getDepthLevel() - dataArray.level, 0});
985 
986  // Store the value (array[index]) into the iterator.
987  generatedCode.addInstruction({FJP::OP_CODE::STO, symbolTable.getDepthLevel() - iterVariable.level, iterVariable.address});
988 
989  // Increment the current index (moving on to the next element).
990  generatedCode.addInstruction({FJP::OP_CODE::LOD, 0, indexAddress});
991  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, 1});
992  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_PLUS});
993  generatedCode.addInstruction({FJP::OP_CODE::STO, 0, indexAddress});
994 
995  // <statement>
996  processStatement();
997 
998  // Jump to the incrementation part of the loop (index++).
999  generatedCode.addInstruction({FJP::OP_CODE::JMP, 0, startForeachBody});
1000 
1001  // Set the address to jump to in case the condition is not satisfied (end of the foreach loop).
1002  generatedCode[exitForeachAddress].m = generatedCode.getSize();
1003 
1004  // Remove the temporary variable (index) off the stack.
1005  generatedCode.addInstruction({FJP::OP_CODE::INC, 0, -1});
1006 }

1137 void FJP::Parser::processCases(Symbol &variable, std::list<int> &breaks) {

1144  token = lexer->getNextToken();

1147  token.tokenType != FJP::TokenType::TRUE &&
1148  token.tokenType != FJP::TokenType::FALSE) {
1149  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_34, ERR_CODE, token.lineNumber);
1150  }

1153  // it is a boolean, convert it into 1/0.
1154  if (token.tokenType == FJP::TokenType::NUMBER) {
1155  token_value = atoi(token.value.c_str());
1156  } else {
1157  token_value = token.value == "true";
1158  }
1159 
1160  // Make sure the literal is the same datatype as the variable used in the switch statement.
1161  if (((variable.symbolType == SYMBOL_INT) && token.tokenType != NUMBER) ||
1162  ((variable.symbolType == SYMBOL_BOOL) && !(token.tokenType == TRUE || token.tokenType == FALSE))) {
1163  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_35, ERR_CODE, token.lineNumber);
1164  }

1167  token = lexer->getNextToken();
1168  if (token.tokenType != FJP::TokenType::COLON) {
1169  FJP::exitProgramWithError(__FUNCTION__, FJP::CompilationErrors::ERROR_11, ERR_CODE, token.lineNumber);
1170  }
1171 
1172  // Compare the literal against the variable used in the switch statement.
1173  generatedCode.addInstruction({FJP::OP_CODE::LOD, symbolTable.getDepthLevel() - variable.level, variable.address});
1174  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, token_value});
1175  generatedCode.addInstruction({FJP::OP_CODE::OPR, 0, FJP::OPRType::OPR_EQ});
1176 
1177  // If it doesn't match up, skip the body of the case statement and jump onto the next case.
1178  int jpc_address = generatedCode.getSize();
1179  generatedCode.addInstruction({FJP::OP_CODE::JPC, 0, 0});
1180

1182  token = lexer->getNextToken();
1183  processStatement();
1184

1331  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, 0});

1503  } else {

1535  generatedCode.addInstruction({FJP::OP_CODE::LDA, symbolTable.getDepthLevel() - symbol.level, 0});

1550  case FJP::TokenType::NUMBER:

1552  generatedCode.addInstruction({FJP::OP_CODE::LIT, 0, numberValue});
1553  break;
1554  // <true/false>
1555  case FJP::TokenType::TRUE:
1556  case FJP::TokenType::FALSE:
